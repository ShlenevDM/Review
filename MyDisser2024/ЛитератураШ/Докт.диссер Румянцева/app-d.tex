\chapter{Вычисление амплитуды рассеяния фотона на электронах замагниченной плазмы 
с излучением двух фотонов}

\label{Sec:app_d}

Определим $S$-матричный элемент
плазменного вклада, просуммированный по всем состояниям 
электронов и позитронов, с учетом соответствующих
функций распределения, следующим образом:
%
\beq
{\cal S}_{pl} = \sum_{\ell, s} \int dn_p \, [{\cal S}_{-} f_{-}(p) + {\cal S}_{+}
f_{+}(p)].
\label{Spl}
\eeq
%
Здесь $dn_p$ - элемент фазового объема
\[
dn_p = \frac{dp_y dp_z L_y L_z}{(2 \pi)^2},
\]
где $L_y, L_z$ - параметры, определяющие объем квантования,
$V = L_x L_y L_z$.
$S$-ма\-трич\-ные элементы ${\cal S}_\mp$ описывают рассеяние фотона
<<вперед>>,  с излучением двух фотонов, на электронах и позитронах соответственно, 
функции $f_{\mp}(p)$  являются соответствующими функциями
распределения. 
%В случае термодинамического равновесия и в системе покоя плазмы они имеют следующий вид
%\[
%f_{\mp}(p) = \frac{1}{e^{\frac{p_0 \mp \mu}{T}}+1}.
%\]
%
 Суммирование в (\ref{Spl}) ведется по спинам и уровням Ландау плазменных
электронов и позитронов.

В пределе сильного поля электроны и позитроны в плазме находятся на
основном уровне Ландау, $\ell = 0$. Поэтому, для нахождения матричных
элементов ${\cal S}_{\mp}$ в этом пределе оказывается удобным пользоваться 
решениями уравнения Дирака~(\ref{eq:psie}) в этом пределе:  
%
\begin{eqnarray}
\Psi^{\eta}_{p,0} (X) = \frac{\beta^{1/4}}{\sqrt{2\sqrt{\pi} E (E+m) L_y L_z}}
e^{-i  (E X_0 - p_y X_2 - p_z X_3)} e^{-\xi^2/2}
u_{\eta}(p_{\mprl}) \, ,
\label{eq:psip1}
\end{eqnarray}
%
\noindent где  $E = \sqrt{p_z^2 + m^2}$.  
Соответствующие биспинорные амплитуды имеют вид
%
\beq
u_{\eta}(p_{\mprl}) =  \left ( 
\begin{array}{c}
(E + m) \phi_{\eta} \\
\eta p_z \phi_{\eta}
\end{array}
\right ),
\label{eq:psim1}
\eeq
%
где
$
\phi_{-} = \left (
\begin{array}{c}
0\\
1
\end{array}
\right )
$ и  $
\phi_{+} = \left (
\begin{array}{c}
1\\
0
\end{array}
\right )
$
описывают состояния электрона и позитрона со спином, направленным против и по полю соответственно.


Используя эти решения, получаем для $S$-матричного элемента, 
соответствующего диаграммам на рис. \ref{fig:DiagGGG}, следующее выражение
%
\beq
{\cal S}_{pl} = \int \frac{dp_y dp_z}{(2 \pi)^2} \, \sum_{i=1}^{6} \, 
[{\cal S}^{(i)}_{-} f_{-}(p) + {\cal S}^{(i)}_{+}
f_{+}(p)],
\label{Spl1}
\eeq
%
\noindent где ${\cal S}^{(i)}_{\mp}$ обозначает матричный элемент 
рассеяния на электронах (позитронах) для $i$-той диаграммы. 
Учитывая, что в процессе рассеяния фотона вперед электроны (позитроны) плазмы не изменяют 
свое состояние (при этом биспиноры, входящие в решения~(\ref{eq:psip1}), образуют  
матрицу плотности), 
запишем, например, ${\cal S}_{-}^{(1)}$ в виде
%
\beq
{\cal S}_{-}^{(1)} &=& -\frac{i e^3 \sqrt{\beta /\pi}}{2 E 
\sqrt{2 \omega V 2 \omega ' V 2 \omega '' V}}\int d^4 x\, d^4 y\, d^4 z\, 
Sp \{[(p\gamma)_{\mprl} + m]\Pi_{-} 
\times \nonumber \\ 
&&\times \ 
(\ee'\gamma) S(z,y)  (\ee''\gamma) S(y,x) (\ee\gamma) \} 
e^{-i[(qx) - (q'z) - (q''y)]} 
\times \nonumber \\ 
&&\times \ 
e^{i(p(z-x))_{\mprl}} e^{-ip_y(z_2-x_2)} 
 e^{-(\beta /2) [(z_1 + p_y /\beta)^2 + (x_1 + p_y /\beta)^2]}.
\label{S1}
\eeq    
%
Здесь $S(x,y)$ - пропагатор электрона в магнитном поле 
(формула~(\ref{eq:Sxy})  приложения~\ref{Sec:app_b}).

Производя в (\ref{S1}) замену переменных $Y = z - y, Z= y - x$, и интегрируя 
по $x$, получим для $\S_{-}^{(1)}$ следующее выражение:

%
\beq
{\cal S}_{-}^{(1)} &=& -\frac{i e^3 (2 \pi)^3 
\delta(\omega - \omega ' - \omega '') \delta(q_y - q'_y - q''_y)
\delta(q_z - q'_z - q''_z)}{2 E 
\sqrt{2 \omega V 2 \omega ' V 2 \omega '' V}}  
\times \label{S2}  \\ 
&\times &\ 
\int d^4 Y\, d^4 Z\, Sp \{[(p\gamma)_{\mprl} + m]\Pi_{-} 
(\ee'\gamma) S(Y)  (\ee''\gamma) S(Z) (\ee\gamma) \} 
\times \nonumber \\ 
&\times &\ 
e^{i[((q'+q'')Z) + (q'Y)]} 
e^{ip_2 (q_x - q'_x - q''_x)/\beta} 
e^{i(p(Y+Z))_{\mprl}} 
 e^{-(\beta /4) [Y^2_{\mprp} + Z^2_{\mprp} + 2(YZ)_{\mprp} + 2i(Y\varphi Z)]},
\nonumber
\eeq    
%

\noindent  где $S(X)$ -- трансляционно-инвариантная часть пропагатора,  
определяемая формулой~(\ref{eq:Sx}) приложения~\ref{Sec:app_b}.

Остальные элементы ${\cal S}_{-}^{(i)}$, соответствующие диаграммам с 
перестановками фотонов, могут быть теперь легко получены из (\ref{S2}). 
Замена $p \to -p$ позволяет найти ${\cal S}_{+}^{(i)}$ - матричные элементы 
рассеяния на позитронах. 
В общем случае вычисление $S$ - матричного элемента 
при произвольных величинах 
импульсов частиц и магнитного поля является довольно сложной задачей. 
Подстановка пропагатора $S(X)$ в формулу (\ref{S2}) приводит к 
чрезвычайно громоздкому выражению в виде двукратного интеграла по 
собственному  времени. Нахождение предела сильного поля из этого выражения 
представляется достаточно трудоемким. Более удобным 
при вычислениях оказывается использовать 
разложение пропагатора электрона по обратной величине магнитного поля. 
Для этого представим трансляционно-инвариантную часть пропагатора 
электрона $\hat S(X)$  следующим образом \cite{Chist:2002mpla}:

\beq
\hat S(X) = \hat S_{-}(X) + \hat S_{+}(X) + \hat S_{\mprp}(X).
\label{Sa_in}
\eeq

\noindent  Здесь  
%
\beq
\hat S_{-}(X)\simeq \frac{i \beta}{2 \pi} 
\exp\left (- \frac{\beta \prp{X}^2}{4} \right ) \int 
\frac{d^2 \pprl{p}}{(2 \pi)^2}\, 
\frac{\pprl{(p \gamma)} + m}{\pprl{p}^2 - m^2}\,\Pi_{-} e^{- i \pprl{(p X)}},	
\eeq
%

\beq
\!\!\!\!\!\!\!\!\!\!\!\!\!\!\!\!\!\!\!\!\!&&\hat S_{+}(X)  \simeq 
-\frac{i}{4 \pi}\, 
[-i\pprl{(\gamma \,\partial/\partial X)} + m]\, 
\pprl{\delta}^2(X) \, \Pi_{+} \, 
\exp \left(\frac{\beta \prp{X}^2}{4} \right)\, 
\Gamma \left (0, \frac{\beta \prp{X}^2}{2} \right),
\nonumber
\\
\\
\!\!\!\!\!\!\!\!\!\!\!\!\!\!\!\!\!\!\!\!\!&&\hat S_{\mprp}(X)  \simeq 
-\frac{1}{2 \pi} \, \pprl{\delta}^2(X)\, \frac{\prp{(X \gamma)}}{\prp{X}^2}
\exp \left(- \frac{\beta \prp{X}^2}{4} \right),
\nonumber
\\
\eeq
%
\noindent где $\pprl{\delta}^2(X) = \delta(X_0)\, \delta(X_3)$,
$\Gamma(a, z)$ - неполная гамма функция:
%
$$
\Gamma(a,z) = \int \limits^\infty_z t^{a - 1} e^{-t} dt.
$$
% 
При этом, как показывает анализ, вклады в первые два члена 
разложения $S$ - матричного элемента по обратной величине 
магнитного поля будут давать  только комбинации:
$\hat S_{-}\hat S_{-}$, $\hat S_{-}\hat S_{+}$, 
$\hat S_{-}\hat S_{\mprp}$, $\hat S_{\mprp}\hat S_{\mprp}$.

Произведя в (\ref{S6}) над всеми 
$\gamma$-матрицами под знаком следа операцию зарядового
сопряжения и используя свойства операторов $\Pi_{\pm}$ из приложения~\ref{Sec:app_e}, 
находим, что след от матриц $\hat{{\cal A}}^{(1)}$ 
может быть преобразован к следу от матриц $\hat{{\cal B}}^{(6)}$ так,
что 
\beq
\mathrm{Sp}\hat{{\cal A}}^{(1)} \, = \, -\mathrm{Sp}\hat{{\cal 
B}}^{(6)*}.
\label{eq:Sp}
\eeq
Путем перестановки фотонов 
 могут быть вычислены остальные $S$-мат\-рич\-ные 
элементы, определяющие вклад плазмы в процесс расщепления фотона.
Используя теперь соотношение (\ref{eq:Sp}) и формулу (\ref{Spl1}), 
легко показать, что все четные по внешнему полю 
(т.е. четные по $\varphi$ и $\tilde \varphi$) слагаемые будут входить в 
$S$-матричный элемент (\ref{Spl1}) в виде нечетных 
по $\mu$ функций, а нечетные -- в виде четных по $\mu$ функций. 
Это утверждение согласуется с 
обобщением теоремы Фарри на случай замагниченной среды, рассмотренным в 
работе \cite{Fradkin}.

Поскольку рассеяние $\gamma e^{\pm} \to \gamma \gamma e^{\pm}$ происходит без
передачи 4-импульса плазме, то этот процесс физически проявляет себя как
расщепление фотона с соответствующим законом сохранения всех
четырех компонент энергии-импульса. Выражение (\ref{S1}),
описывающее рассеяние на одном электроне, содержит только три
$\delta$-функции. Однако в $S$-матричном элементе (\ref{Spl1}),
учитывающем рассеяние на всех электронах и позитронах плазмы, содержится
интеграл по обобщенному импульсу $p_y$, который дает недостающую
$\delta$-функцию.
Тогда амплитуда, соответствующая плазменному вкладу, может быть определена
обычным образом
%
\beq
