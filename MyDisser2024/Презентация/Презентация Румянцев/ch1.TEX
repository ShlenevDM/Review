\begin{Slide}{Введение} 

\vspace*{5mm}

\begin{center}

\centerline{\darkgreen Два предельных случая,}

\vspace*{5mm}

\noindent
в зависимости от соотношения между полевым инвариантом: 
%
$$e^2 F_{\mu \nu} F^{\nu \mu} \equiv e^2 (F F) = - 2 e^2 B^2$$

\noindent
и динамическим инвариантом:
%
$$e^2 p_\mu F^{\mu \nu} F_{\nu \rho} p^\rho \equiv e^2 (p F F p) 
= e^2 B^2 E^2 \sin^2 \theta, $$
%
где $\theta$ -- угол между импульсом частицы $\bf p$ и направлением 
поля $\bf B$. 

В обезразмеренном виде:
%
$$\chi^2 = \frac{e^2 (p F F p)}{m_e^6} .$$

\end{center}

\end{Slide}

\begin{Slide}{Введение} 

\begin{center}

\vspace*{5mm}

\centerline{\blue \it Предел сильного поля, $e B \gg E^2 \, (T^2, \mu^2)$}

\vspace*{5mm}

В инвариантной форме:
%
$$\left[e^2 (F F)\right]^{3/2} \gg e^2 (p F F p) .$$
%
Электроны находятся только на основном уровне Ландау.
Техника вычислений -- ``двумерная электродинамика'' -- 
В.В.~Скобелев, Ю.М.~Лоскутов (1975-76). 
Развитие -- ковариантная техника -- Н.В.~Михеев и др. (1996-99).


\vspace*{5mm}

\centerline{\blue \it Предел относительно слабого поля, $E^2 \gg e B$.}

\vspace*{5mm}

В инвариантной форме:
%
$$\left[e^2 (F F)\right]^{3/2} \ll e^2 (p F F p) .$$
%
Это условие автоматически выполняется в случае скрещенного поля, 
где $(F F) = 0$. 
Техника вычислений в скрещенном поле -- А.И.~Никишов, В.И.~Ритус (1964).

\end{center}

\end{Slide}

\begin{Slide}{Нейтринное рождение лептонных пар...} 

\begin{center}

{\red\LARGE Нейтринное рождение лептонных пар  
                 во внешнем электромагнитном поле}
\end{center}

\vspace*{5mm}

\begin{center}

\centerline{\darkgreen Вероятность процесса $\nu \to \nu e^- e^+$ 
в пределе $\chi \gg 1$} 

\begin{eqnarray}
W (\nu \to \nu e^- e^+) \, = \, {\red K} \, W_0 \,\chi^2 
\, \left(\ln \chi - {1 \over 2} \ln 3 - \gamma_E + {\red \Delta} \right),
\nonumber
\end{eqnarray}
%
где 
%
\begin{eqnarray}
W_0 \, = \, 
\frac{G_F^2 \,(C_V^2 + C_A^2) \,m_e^6}
{27 \pi^3 E},
\nonumber
\end{eqnarray}
%
$\gamma_E = 0.577 \dots$ -- постоянная Эйлера, 
$E$ -- энергия начального нейтрино.

$$C_V = \pm 1/2 + 2 \sin^2 \theta_W, \, C_A = \pm 1/2.$$

{\blue Формула для вероятности справедлива при $\ln \chi \gg 1$,}

{\red но возможна ситуация:
%
 $$\chi \gg 1, \quad \mbox{однако} \quad \ln \chi \sim 1.$$ }

\end{center}

\end{Slide}

\nonumber
\end{eqnarray}
