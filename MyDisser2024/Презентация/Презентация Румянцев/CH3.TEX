\begin{Slide}{Расщепление фотона в сильно замагниченной плазме} 

\begin{center}

\vspace*{15mm}

\centerline{\darkgreen Вероятность расщепления фотона}

\vspace*{10mm}

\begin{eqnarray}
&&W_{\lambda \to \lambda^{'} \lambda^{''}} = 
\frac{g_{\lambda' \lambda''}}
{32 \pi^2 \omega}
\int \mid {\cal M_{\lambda \lambda^{'} \lambda^{''}}}\mid^2
Z_{\lambda}Z_{\lambda^{'}}Z_{\lambda^{''}} \times
\nonumber \\
&&\times (1 + f_{\omega'})(1 + f_{\omega''}) 
\delta (\omega_{\lambda}({\bf k}) - 
\omega_{\lambda^{'}}({\bf k} - {\bf k^{''}}) - \omega_{\lambda^{''}}({\bf k^{''}})) 
\frac{d^3 k^{''}}{\omega_{\lambda^{'}} \omega_{\lambda^{''}}},
\nonumber
\end{eqnarray}                    

\vspace*{5mm}

\noindent где $f_{\omega} = (e^{\omega/T} - 1)^{-1}$ -- функция распределения 
фотонов, множитель $g_{\lambda' \lambda''} = 
1 - (1/2)\delta_{\lambda' \lambda''}$ введен 
для учета возможной тождественности фотонов в конечном состоянии.

\end{center}

\end{Slide}
